CREATE VIEW v_kohteet_ilman_rakennuksia AS (
 SELECT k.id,
    k.nimi,
    k.geom,
    k.alkupvm,
    k.loppupvm,
    k.voimassaolo,
    k.kohdetyyppi_id
   FROM jkr.kohde k

           FROM jkr.kohteen_rakennukset kr
          WHERE (k.id = kr.kohde_id))))
